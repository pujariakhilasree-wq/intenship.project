import random

# Predefined responses
responses = {
    "greeting": ["hello", "hi", "hey"],
    "name": ["what is your name", "who are you"],
    "course": ["courses", "what courses", "available courses"],
    "fees": ["fees", "price", "cost"],
    "bye": ["bye", "exit", "quit"]
}

# Reply messages
replies = {
    "greeting": ["Hello! 👋", "Hi there!", "Hey! How can I help you?"],
    "name": ["I am your AI chatbot 🤖", "You can call me SmartBot!"],
    "course": ["We offer AI, ML, and Python courses."],
    "fees": ["Fees vary depending on the course. Please contact admin."],
    "bye": ["Goodbye! 😊", "See you later!", "Bye! Have a great day!"],
    "default": ["Sorry, I didn’t understand that.", "Can you please rephrase?"]
}

def chatbot():
    print("🤖 SmartBot: Hello! Type 'bye' to exit.\n")

    while True:
        user_input = input("You: ").lower()

        found = False

        for key, keywords in responses.items():
            for word in keywords:
                if word in user_input:
                    print("🤖 SmartBot:", random.choice(replies[key]))
                    found = True

                    if key == "bye":
                        return
                    break
            if found:
                break

        if not found:
            print("🤖 SmartBot:", random.choice(replies["default"]))

# Run chatbot
chatbot()
